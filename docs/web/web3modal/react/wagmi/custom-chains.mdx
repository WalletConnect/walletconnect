# Custom Chains

## Custom wagmi chains

<CustomWagmiChainsPartial />

```tsx
import { w3mProvider } from '@web3modal/ethereum'
import { configureChains } from 'wagmi'
import { arbitrum, mainnet, avalanche } from 'wagmi/chains'

const chains = [mainnet, avalanche, arbitrum]
const { publicClient } = configureChains(chains, [
  w3mProvider({
    /* ... */
  })
])
```

## Custom default chain

<CustomDefaultChainPartial />

```tsx
import { optimism } from 'wagmi/chains'

function App() {
  return <Web3Modal defaultChain={optimism} />
}
```

## Custom chain providers

<CustomChainProvidersPartial />

```tsx
import { mainnet } from 'wagmi/chains'
import { infuraProvider } from 'wagmi/providers/infura'
import { w3mProvider } from '@web3modal/ethereum'

const { chains, publicClient } = configureChains(
  [mainnet],
  [
    w3mProvider({
      /* ... */
    }),
    alchemyProvider({
      /* ... */
    })
  ]
)
```
